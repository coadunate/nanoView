# INSTRUCTIONS FOR DEPLOYING NANOVIEW# Clone the nanoView project.$ git clone https://github.com/coadunate/nanoView.git# Setting up and building the project.$ cd nanoView$ npm install$ npm run build# This needs to be run on separate shell window (make sure that you are in the nanoView directory)$ python3 ./nanopore-api/app.py# This also needs to be run on a separate shell window. (make sure that you are in nanoView directory)$ npm run http-server# Go to 127.0.0.1:8080/examples to view the nanoView.
